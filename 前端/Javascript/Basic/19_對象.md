- 添加屬性
	- `對象名.屬性名 = 屬性值`
- 函數也可是對象的屬性
- 使用`for in`遍歷對象
	- 
		```js
		for (const key in obj){}
		```
		- `key` 返回屬性名
		- `obj[key]` 的到屬性值

```js
        var obj = new Object();

        // 向對象添加屬性
        obj.name = 'wang';

        // 函數也可是對象的屬性
        obj.sayName = function () {
            console.log('obj.name: ', obj.name);            
        }
        obj.sayName;
        
        let obj2 = {
            nama: 'wang',
            age: 5,
            sayName: function () {
                console.log(this.nama);
            }
        }
        for (const key in obj2) {
            console.log('key: ', key);              // 得到屬性名
            console.log('obj2[key]: ', obj2[key]);    // 得到屬性值
        }
```
